\makeatletter
\let\oldlt\longtable
\let\endoldlt\endlongtable

\def\longtable{\@ifnextchar[\longtable@i \longtable@ii}
\def\longtable@i[#1]{%
  % Temporarily fool longtable into thinking we're in onecolumn mode
  \@twocolumnfalse
  \oldlt[#1]%
}
\def\longtable@ii{%
  % Temporarily fool longtable into thinking we're in onecolumn mode  
  \@twocolumnfalse
  \oldlt
}
\def\endlongtable{%
  \endoldlt
  % Restore twocolumn flag
  \@twocolumntrue
}
\makeatother
